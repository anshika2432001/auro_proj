1) Grade

SELECT DISTINCT grade FROM student_master WHERE grade BETWEEN 1 AND 12 ORDER BY grade;

2) Subject

SELECT DISTINCT subject FROM exam_details ORDER BY subject;

3) School Location

SELECT DISTINCT school_location, 
CASE 
   WHEN school_location = 1 THEN 'Rural'
   WHEN school_location = 2 THEN 'Urban'
   ELSE 'Unknown'
END AS location_name
FROM student_extra_data
ORDER BY school_location;


4) State

SELECT state_id, state_name FROM state_master ORDER BY state_id;

5) District

SELECT district_id, district_name, state_id FROM state_district_master ORDER BY district_id;

6) Social Group

SELECT DISTINCT social_group, 
CASE 
   WHEN social_group = '1' THEN 'SC'
   WHEN social_group = '2' THEN 'ST'
   WHEN social_group = '3' THEN 'OBC'
   WHEN social_group = '4' THEN 'General'
   WHEN social_group = '5' THEN 'Other'
   ELSE 'Unknown'
END AS social_group_name
FROM student_extra_data
ORDER BY social_group;

7) Gender

    gender, 
    CASE 
        WHEN gender = 'M' THEN 'Male'
        WHEN gender = 'F' THEN 'Female'
        WHEN gender = 'O' THEN 'Other'
        WHEN gender = '' THEN 'Prefer Not to say'
        ELSE 'Unknown'
    END AS gender_name
FROM 
    student_demographic
WHERE 
    gender IS NOT NULL
ORDER BY 
    gender;


8) Age Group

SELECT DISTINCT 
    CASE 
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) <= 6 THEN 'Upto 6'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 7 AND 10 THEN '6-10'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 11 AND 13 THEN '11-13'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 14 AND 15 THEN '14-15'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 16 AND 17 THEN '16-17'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) > 17 THEN '>17'
    END AS age_group
FROM 
    student_demographic
WHERE 
    dob IS NOT NULL
    AND CASE 
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) <= 6 THEN 'Upto 6'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 7 AND 10 THEN '6-10'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 11 AND 13 THEN '11-13'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 14 AND 15 THEN '14-15'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) BETWEEN 16 AND 17 THEN '16-17'
        WHEN TIMESTAMPDIFF(YEAR, dob, CURDATE()) > 17 THEN '>17'
        ELSE NULL
    END IS NOT NULL
ORDER BY 
    age_group;

9) Education Board

SELECT DISTINCT education_board, 
CASE 
   WHEN education_board = '1' THEN 'CBSE'
   WHEN education_board = '2' THEN 'State Board'
   WHEN education_board = '3' THEN 'ICSE'
   WHEN education_board = '4' THEN 'International Board'
   WHEN education_board = '5' THEN 'Others'
   WHEN education_board = '6' THEN 'Both CBSE and State Board'
   ELSE 'Unknown'
END AS education_board_name
FROM student_demographic
ORDER BY education_board;

10) School Management Type

SELECT DISTINCT school_management, 
CASE 
   WHEN school_management = '1' THEN 'Department of Education'
   WHEN school_management = '2' THEN 'Tribal/Social Welfare Department'
   WHEN school_management = '3' THEN 'Local body'
   WHEN school_management = '4' THEN 'Pvt. Aided'
   WHEN school_management = '5' THEN 'Pvt. Unaided'
   WHEN school_management = '6' THEN 'Others'
   WHEN school_management = '7' THEN 'Central Govt.'
   WHEN school_management = '8' THEN 'Unrecognized'
   WHEN school_management = '97' THEN 'Madarsa recognized (by Wakf board/Madarsa Board)'
   WHEN school_management = '98' THEN 'Madarsa unrecognized'
   ELSE 'Unknown'
END AS school_management_name
FROM student_demographic
ORDER BY school_management;
